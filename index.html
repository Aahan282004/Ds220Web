# Let's start with the data cleaning process.

# Convert 'Postal Code', 'Legislative District', and '2020 Census Tract' to strings to prevent any future analysis issues.
df['Postal Code'] = df['Postal Code'].astype(str).str[:-2].str.pad(5, fillchar='0')  # Remove the '.0' and pad if necessary
df['Legislative District'] = df['Legislative District'].astype(str).str.split('.').str[0]  # Convert to string, remove decimal
df['2020 Census Tract'] = df['2020 Census Tract'].astype(str)

# Now let's handle the missing values. For simplicity, we'll fill missing values for 'County', 'City', 'Vehicle Location', 
# and 'Electric Utility' with the string 'Unknown'. Since 'Postal Code' is critical for location-based analysis, rows with missing 
# postal codes will be removed.
df['County'].fillna('Unknown', inplace=True)
df['City'].fillna('Unknown', inplace=True)
df['Vehicle Location'].fillna('Unknown', inplace=True)
df['Electric Utility'].fillna('Unknown', inplace=True)

# Drop rows where 'Postal Code' is missing
df.dropna(subset=['Postal Code'], inplace=True)

# Check for duplicates based on 'DOL Vehicle ID' as it should be unique and drop them if any.
df.drop_duplicates(subset=['DOL Vehicle ID'], inplace=True)

# Re-check the missing values and the data types after cleaning
missing_values_cleaned = df.isnull().sum()
data_types_cleaned = df.dtypes
